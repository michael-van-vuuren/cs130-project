import convert from 'convert-units'
/**
 * A utility class for converting ingredient units to SI.
 *
 * @class UnitConverter
 */
class UnitConverter {
    /**
     * Converts the units of an ingredient to SI.
     *
     * @static
     * @param {Object} ingredient - The ingredient to be converted.
     * @param {string} ingredient.name - The name of the ingredient.
     * @param {number} ingredient.id - The unique identifier for the ingredient.
     * @param {number} ingredient.quantity - The quantity of the ingredient.
     * @param {string} ingredient.units - The original units of measurement for the quantity.
     * @returns {Object} - Returns the ingredient with units converted to SI.
     */
    static convertToSI(ingredient) {

        const { name, id, quantity, units } = ingredient

        if (units === 'g' || units === 'ml' || units === 'count')
            return ingredient

        try {
            let convertedQuantity

            // volume
            if (['l', 'cup', 'gal', 'fl-oz', 'qt', 'pnt', 'tsp', 'Tbs'].includes(units)) {
                convertedQuantity = parseInt(convert(quantity).from(units).to('ml'))
                return { name: name, id: id, quantity: convertedQuantity, units: 'ml' }
            }
            // mass
            else if (['oz', 'lb', 'kg', 'mg'].includes(units)) {
                convertedQuantity = parseInt(convert(quantity).from(units).to('g'))
                return { name: name, id: id, quantity: convertedQuantity, units: 'g' }
            }
            else {
                return ingredient
            }
        } catch (error) {
            console.error(`error converting unit: ${error.message}`)
            return ingredient
        }
    }

    /**
     * Converts the units of ingredients in the pantry to the International System of Units (SI).
     *
     * @static
     * @param {Array} pantry - The list of ingredients in the pantry.
     * @returns {Array} - Returns a new array with ingredients in the pantry converted to SI units.
     */
    static convertPantryToSI(pantry) {
        return pantry.map((ingredient) => this.convertToSI(ingredient))
    }
}

export default UnitConverter
