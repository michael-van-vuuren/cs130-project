api/database.route.js

import express from 'express';
import PantryCtrl from './pantry.controller.js';
import IngredientCtrl from './ingredient.controller.js';
import GoogleLoginCtrl from './google-login.controller.js';
import LLMCtrl from './llm.controller.js';
import RecipeCtrl from './recipe.controller.js';

/**
 * Express router instance for FlavorFinder API.
 * @type {router}
 */
const router = express.Router();

/**
 * Route to create a new user in the FlavorFinder pantries.
 * @name POST /api/users/new
 * @function
 * @memberof router
 * @inner
 */
router.route('/users/new')
  .post(PantryCtrl.apiAddUser);

/**
 * Route to check if a user already exists in the FlavorFinder pantries.
 * @name GET /api/users/exists/:sub
 * @function
 * @memberof router
 * @inner
 */
router.route('/users/exists/:sub')
  .get(PantryCtrl.apiUserExists);

/**
 * Routes to get, update, and delete items from a user's pantry in the FlavorFinder pantries.
 * @name GET /api/users/:id
 * @function
 * @memberof router
 * @inner
 * @name PUT /api/users/:id
 * @function
 * @memberof router
 * @inner
 * @name DELETE /api/users/:id
 * @function
 * @memberof router
 * @inner
 */
router.route('/users/:id')
  .get(PantryCtrl.apiGetPantry)
  .put(PantryCtrl.apiUpdatePantry)
  .delete(PantryCtrl.apiDeletePantryItem);

/**
 * Route to get Google Login API client ID.
 * @name GET /api/clientid
 * @function
 * @memberof router
 * @inner
 */
router.route('/clientid')
  .get(GoogleLoginCtrl.apiGetClientId);

/**
 * Route to get ingredients for the pantry, including name and image.
 * @name GET /api/ingredients
 * @function
 * @memberof router
 * @inner
 */
router.route('/ingredients')
  .get(IngredientCtrl.serverGetIngredients);

/**
 * Route to get the expanded form of a specific ingredient.
 * @name GET /api/ingredient/:id
 * @function
 * @memberof router
 * @inner
 */
router.route('/ingredient/:id')
  .get(IngredientCtrl.serverGetIngredient);

/**
 * Route for communication between the chatbot and the LLM (Language Model).
 * @name POST /api/chatbot
 * @function
 * @memberof router
 * @inner
 */
router.route('/chatbot')
  .post(LLMCtrl.serverSendMessage);

/**
 * Route to get the recipe pool based on the provided user ID and threshold.
 * @name GET /api/recipe-pool/:id/:threshold
 * @function
 * @memberof router
 * @inner
 */
router.route('/recipe-pool/:id/:threshold')
  .get(RecipeCtrl.serverGetRecipePool);

export default router;
