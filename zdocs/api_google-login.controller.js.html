api/google-login.controller.js

import { Request, Response, NextFunction } from 'express';
import GoogleLoginDAO from '../dao/google-loginDAO.js';

/**
 * Controller class for handling API functionality.
 * @class GoogleLoginController
 * @api
 */
class GoogleLoginController {
  /**
   * Retrieves the Google client ID for Google login functionality.
   *
   * @static
   * @async
   * @function
   * @param {Response|Error} res - The Google clientid.
   * @return - Returns null if Google clientid not found.
   */
  static async apiGetClientId(res) {
    try {
      let clientid = await GoogleLoginDAO.getClientId();
      if (!clientid) {
        res.status(404).json({ error: 'cannot find Google clientid' });
        return;
      }
      res.json(clientid);
    } catch (e) {
      res.status(500).json({ error: e.message });
    }
  }
}

export default GoogleLoginController
