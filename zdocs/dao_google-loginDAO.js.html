import { MongoClient, Collection, ObjectId } from 'mongodb';

/**
 * Data Access Object (DAO) class for Google Login functionality.
 * @class
 */
class GoogleLoginDAO {
  /**
   * Injects the MongoDB connection for the GoogleLoginDAO class.
   *
   * @static
   * @async
   * @function
   * @param {MongoClient} conn - The MongoDB connection.
   * @returns {Promise<void>} A Promise that resolves once the injection is complete.
   * @throws {Error} If an error occurs during the operation.
   */
  static async injectDB(conn) {
    if (googlelogindb) {
      return;
    }
    try {
      googlelogindb = await conn.db('flavorfinder').collection('google-login-api');
    } catch (e) {
      console.error(`Unable to establish collection handles in GoogleLoginDAO: ${e}`);
      throw e;
    }
  }

  /**
   * Retrieves the new user MongoDB _id.
   *
   * @static
   * @async
   * @function
   * @returns {Promise<Object|null>} The Google client ID
   * @throws {Error} If an error occurs during the operation.
   */
  static async getClientId() {
    try {
      const result = await googlelogindb.findOne(
        { _id: new ObjectId('655d595d868e371db86da999') }
      );

      return result;
    } catch (e) {
      console.error(`Unable to get Google client id: ${e}`);
      throw e;
    }
  }
}

export default GoogleLoginDAO
